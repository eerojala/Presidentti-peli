1   package presidenttipeli.logiikka.luojat;
2   
3   import java.util.ArrayList;
4   import presidenttipeli.domain.Nappula;
5   import presidenttipeli.domain.Pelaaja;
6   import presidenttipeli.util.KuvanLataaja;
7   
8   /**
9    * Luojaluokka joka luo pelille nappulat.
10   */
11  public class NappuloidenLuoja extends Luoja {
12  
13      private ArrayList<Pelaaja> pelaajat;
14      private ArrayList<Nappula> nappulat;
15  
16      public NappuloidenLuoja(ArrayList<Pelaaja> pelaajat) {
17          this.pelaajat = pelaajat;
18          nappulat = new ArrayList();
19      }
20  
21      @Override
22      public void luo() {
23          for (Pelaaja pelaaja : pelaajat) {
24              Nappula nappula = new Nappula(pelaaja);
25              pelaaja.setNappula(nappula);
26              nappulat.add(nappula);
27          }
28          asetaNappuloilleKuvat();
29      }
30  
31      private void asetaNappuloilleKuvat() {
32          for (int i = 0; i < nappulat.size(); i++) {
33              String tiedostopolku = "/kuvat/Pelaaja";
34              tiedostopolku += (i + 1) + ".png";
35              nappulat.get(i).setKuva(KuvanLataaja.palautaKuva(tiedostopolku));
36          }
37      }
38  
39      public ArrayList<Nappula> getNappulat() {
40          return nappulat;
41      }
42  
43  }
