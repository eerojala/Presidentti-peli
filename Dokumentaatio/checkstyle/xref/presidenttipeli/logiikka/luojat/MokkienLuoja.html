1   package presidenttipeli.logiikka.luojat;
2   
3   import java.util.ArrayDeque;
4   import java.util.ArrayList;
5   import java.util.Collections;
6   import java.util.Scanner;
7   import presidenttipeli.domain.Mokki;
8   
9   /**
10   * Luojaluokka joka luo pelille mökit.
11   */
12  public class MokkienLuoja extends Luoja {
13  
14      private ArrayList<Mokki> lista;
15  
16      public MokkienLuoja() {
17          lista = new ArrayList();
18      }
19  
20      @Override
21      public void luo() {
22          asetaScannerilleTiedosto("tekstit/Mokit.txt",
23                  MokkienLuoja.class.getClassLoader());
24          luoMokit(lukija);
25      }
26  
27      private void luoMokit(Scanner lukija) {
28          boolean arvoViimeksi = true;
29          String nimi = "";
30          int arvo = 0;
31  
32          while (lukija.hasNext()) {
33              String rivi = lukija.nextLine();
34              if (!rivi.isEmpty()) {
35                  if (arvoViimeksi) {
36                      nimi = rivi;
37                      arvoViimeksi = false;
38                  } else {
39                      arvo = Integer.parseInt(rivi);
40                      arvoViimeksi = true;
41                  }
42              } else {
43                  lista.add(new Mokki(nimi, arvo));
44              }
45          }
46      }
47  
48      public ArrayList<Mokki> getLista() {
49          return lista;
50      }
51  
52      /**
53       * Sekoittaa mökkilistan ja palauttaa sen ArrayDequena
54       *
55       * @return Sekoitetut mökit ArrayDequena.
56       */
57      public ArrayDeque<Mokki> getSekoitetutMokit() {
58          Collections.shuffle(lista);
59          ArrayDeque<Mokki> mokit = new ArrayDeque();
60          for (Mokki mokki : lista) {
61              mokit.add(mokki);
62          }
63          return mokit;
64      }
65  
66  }
