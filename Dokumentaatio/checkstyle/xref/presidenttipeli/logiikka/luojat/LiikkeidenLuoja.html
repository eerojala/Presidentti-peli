1   package presidenttipeli.logiikka.luojat;
2   
3   import java.util.ArrayDeque;
4   import java.util.ArrayList;
5   import java.util.Collections;
6   import java.util.Scanner;
7   import presidenttipeli.domain.Liike;
8   
9   /**
10   * Luojaluokka joka luo pelille liikkeet.
11   */
12  public class LiikkeidenLuoja extends Luoja {
13  
14      private ArrayList<Liike> lista;
15  
16      public LiikkeidenLuoja() {
17          lista = new ArrayList();
18      }
19  
20      @Override
21      public void luo() {
22          asetaScannerilleTiedosto("tekstit/Liikkeet.txt",
23                  LiikkeidenLuoja.class.getClassLoader());
24          luoLiikkeet(lukija);
25      }
26  
27      private void luoLiikkeet(Scanner lukija) {
28          int laskuri = 1;
29          Liike liike = new Liike("Temp", "Temp", 0, 0);
30  
31          while (lukija.hasNext()) {
32              String rivi = lukija.nextLine();
33              if (!rivi.isEmpty()) {
34                  if (laskuri == 1) {
35                      liike.setTyyppi(rivi);
36                      laskuri++;
37                  } else if (laskuri == 2) {
38                      liike.setNimi(rivi);
39                      laskuri++;
40                  } else if (laskuri == 3) {
41                      liike.setArvo(Integer.parseInt(rivi));
42                      laskuri++;
43                  } else {
44                      liike.setTuotto(Integer.parseInt(rivi));
45                      laskuri = 1;
46                  }
47              } else {
48                  lista.add(liike);
49                  liike = new Liike("Temp", "Temp", 0, 0);
50              }
51          }
52      }
53  
54      public ArrayList<Liike> getLista() {
55          return lista;
56      }
57  
58      /**
59       * Sekoittaa liikelistan ja palauttaa sen ArrayDequena.
60       *
61       * @return Sekoitetut liikkeet ArrayDeque-muodossa.
62       */
63      public ArrayDeque<Liike> getSekoitetutLiikkeet() {
64          ArrayDeque<Liike> liikkeet = new ArrayDeque();
65          Collections.shuffle(lista);
66          for (Liike liike : lista) {
67              liikkeet.add(liike);
68          }
69          return liikkeet;
70      }
71  
72  }
