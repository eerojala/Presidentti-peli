1   package presidenttipeli.logiikka.luojat;
2   
3   import java.util.ArrayDeque;
4   import java.util.ArrayList;
5   import java.util.Collections;
6   import presidenttipeli.domain.Ammatti;
7   
8   /**
9    * Luojaluokka joka luo sattuma-ammatit pelille.
10   */
11  public class SattumaAmmattienLuoja extends AmmatinLuoja {
12  
13      private ArrayList<Ammatti> ammatit;
14  
15      public SattumaAmmattienLuoja() {
16          ammatit = new ArrayList();
17      }
18  
19      @Override
20      public void luo() {
21          asetaScannerilleTiedosto("tekstit/SattumaAmmatit.txt",
22                  SattumaAmmattienLuoja.class.getClassLoader());
23          luoAmmatit(lukija, false, ammatit);
24      }
25  
26      public ArrayList<Ammatti> getAmmatit() {
27          return ammatit;
28      }
29  
30      /**
31       * Sekoittaa ammattilistan ja palauttaa sen ArrayDequena.
32       *
33       * @return Sekoitetut ammatit ArrayDequena.
34       */
35      public ArrayDeque<Ammatti> getSekoitetutAmmatit() {
36          return luoPino();
37      }
38  
39      private ArrayDeque<Ammatti> luoPino() {
40          ArrayDeque<Ammatti> palautettava = new ArrayDeque();
41          Collections.shuffle(ammatit);
42          for (Ammatti ammatti : ammatit) {
43              palautettava.add(ammatti);
44          }
45          return palautettava;
46      }
47  
48  }
