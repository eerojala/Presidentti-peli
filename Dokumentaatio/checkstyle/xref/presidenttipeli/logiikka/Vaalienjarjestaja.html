1   package presidenttipeli.logiikka;
2   
3   import java.util.ArrayList;
4   import presidenttipeli.domain.Pelaaja;
5   import presidenttipeli.domain.Pelilauta;
6   import presidenttipeli.domain.tapahtumat.Vaalit;
7   import presidenttipeli.logiikka.luojat.TapahtumienLuoja;
8   
9   public class Vaalienjarjestaja {
10  
11      /**
12       * Logiikkaluokka joka huolehtii eduskunta- ja presidentinvaalien
13       * järjestämisestä.
14       */
15      private TapahtumienLuoja luoja;
16      private ArrayList<Integer> saadutAanet;
17  
18      public Vaalienjarjestaja(Pelilauta lauta) {
19          luoja = new TapahtumienLuoja(lauta);
20          saadutAanet = new ArrayList();
21      }
22  
23      /**
24       * Järjestää kansanedustajanvaalit. Jos vaalit onnistuvat pelaajasta tulee
25       * kansanedustaja.
26       *
27       * @param tarvittavaAanimaara Tarvittava äänimäärä kansanedustajan
28       * pääsemiseksi.
29       * @param pelaaja Pelaaja joka yrittää vaaleja.
30       */
31      public void jarjestaEduskuntavaalit(int tarvittavaAanimaara, Pelaaja pelaaja) {
32          Vaalit vaalit = luoja.luoVaalit(tarvittavaAanimaara);
33          if (vaalit.suoritaTapahtuma(pelaaja) == true) {
34              luoja.luoKansanedustajuuteenVaikuttavaTapahtuma(true).suoritaTapahtuma(pelaaja);
35          }
36          saadutAanet = vaalit.getSaadutAanet();
37      }
38  
39      /**
40       * Järjestää presidnetinvaalit. Jos pelaaja voittaa vaaleissa voittaa hän
41       * samalla koko pelin.
42       *
43       * @param tarvittavaAanimaara Tarvittava äänimäärä presidentiksi
44       * pääsemiseksi.
45       * @param pelaaja Pelaaja joka yrittää vaaleja.
46       *
47       * @return True jos pelaaja voitti vaalit, muuten false.
48       */
49      public boolean jarjestaPresidentinvaalit(int tarvittavaAanimaara, Pelaaja pelaaja) {
50          Vaalit vaalit = luoja.luoVaalit(tarvittavaAanimaara);
51          boolean onnistuiko = false;
52          if (vaalit.suoritaTapahtuma(pelaaja) == true) {
53              onnistuiko = true;
54          }
55          saadutAanet = vaalit.getSaadutAanet();
56          return onnistuiko;
57      }
58  
59      public ArrayList<Integer> getSaadutAanet() {
60          return saadutAanet;
61      }
62  
63      /**
64       *  Palauttaa saadut yksittäiset äänet summattuna.
65       * 
66       *  @return Saadut yksittäiset äänet summattuna.
67       */
68      public int getSaadutAanetSummattuna() {
69          int summa = 0;
70          for (Integer aani : saadutAanet) {
71              summa += aani; 
72          }
73          return summa;
74      }
75  
76  }
