1   package presidenttipeli.logiikka;
2   
3   import presidenttipeli.domain.Liike;
4   import presidenttipeli.domain.Mokki;
5   import presidenttipeli.domain.Pelaaja;
6   import presidenttipeli.domain.Pelilauta;
7   import presidenttipeli.logiikka.luojat.TapahtumienLuoja;
8   
9   /**
10   * Logiikkaluokka joka huolehtii mökkien ja liikkeiden hankkimisesta ja
11   * myymisestä.
12   */
13  public class Kiinteistonvalittaja {
14  
15      private Pelilauta lauta;
16      private TapahtumienLuoja luoja;
17      private Pankinjohtaja johtaja;
18  
19      public Kiinteistonvalittaja(Pelilauta lauta, Pankinjohtaja johtaja) {
20          this.lauta = lauta;
21          luoja = new TapahtumienLuoja(lauta);
22          this.johtaja = johtaja;
23      }
24  
25      /**
26       * Antaa pelaajalle mökin laudalta ja veloittaa mökin arvon verran rahaa
27       * pelaajalta
28       *
29       * @param pelaaja Pelaaja jolle mökki annetaan.
30       * @param kerroin Kerroin joka vaikuttaa pelaajalta otettavaan rahaan.
31       */
32      public void annaPelaajalleMokki(Pelaaja pelaaja, double kerroin) {
33          Mokki mokki = lauta.getMokit().poll();
34          double arvo = mokki.getArvo() * kerroin;
35          johtaja.otaPelaajaltaRahaa(pelaaja, (int) arvo);
36          luoja.luoMokkeihinVaikuttavaTapahtuma(mokki, true).suoritaTapahtuma(pelaaja);
37      }
38  
39      /**
40       * Ottaa pelaajalta mökin, vie sen takaisin laudalle ja antaa mökin arvon
41       * verran rahaa pelaajalle.
42       *
43       * @param pelaaja Pelaaja jolta mökki otetaan pois.
44       * @param mokki Mökki joka otetaan pois.
45       * @param kerroin Kerroin joka vaikuttaa pelaajalle annettavaan rahaan.
46       */
47      public void otaPelaajaltaMokkiPois(Pelaaja pelaaja, Mokki mokki, double kerroin) {
48          double arvo = mokki.getArvo() * kerroin;
49          johtaja.annaPelaajalleRahaa(pelaaja, (int) arvo);
50          luoja.luoMokkeihinVaikuttavaTapahtuma(mokki, false).suoritaTapahtuma(pelaaja);
51      }
52  
53      /**
54       * Antaa pelaajalle liikkeen laudalta ja veloittaa liikkeen arvon verran
55       * rahaa pelaajalta.
56       *
57       * @param pelaaja Pelaaja jolle liike annetaan
58       * @param kerroin Kerroin joka vaikuttaa pelaajalta otettavaan rahaan.
59       */
60      public void annaPelaajalleLiike(Pelaaja pelaaja, double kerroin) {
61          Liike liike = lauta.getLiikkeet().poll();
62          double arvo = liike.getArvo() * kerroin;
63          johtaja.otaPelaajaltaRahaa(pelaaja, (int) arvo);
64          luoja.luoLiikkeisiinVaikuttavaTapahtuma(liike, true).suoritaTapahtuma(pelaaja);
65      }
66  
67      /**
68       * Ottaa pelaajalta liikkeen pois, vie sen takaisin laudalle ja antaa
69       * pelaajalle liikkeen arvon verran rahaa.
70       *
71       * @param pelaaja Pelaaja jolta liike otetaan pois
72       * @param liike Liike joka otetaan pois
73       * @param kerroin Kerroin joka vaikuttaa pelaajalle annettavaan rahaan.
74       */
75      public void otaPelaajaltaLiikePois(Pelaaja pelaaja, Liike liike, double kerroin) {
76          double arvo = liike.getArvo() * kerroin;
77          johtaja.annaPelaajalleRahaa(pelaaja, (int) arvo);
78          luoja.luoLiikkeisiinVaikuttavaTapahtuma(liike, false).suoritaTapahtuma(pelaaja);
79      }
80  }
