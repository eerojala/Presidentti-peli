5   /**
6    * Tapahtumaluokka joka veloittaa pelaajalta rahaa hänen omistamistaan mökeistä ja
7    * liikkeistä.
8    */
9   public class MaksaKulujaLiikkeistaJaMokeista implements Tapahtuma {
10  
11      private int maksettavaHinta;
12  
13      public MaksaKulujaLiikkeistaJaMokeista(int maksettavaHinta) {
14          this.maksettavaHinta = maksettavaHinta;
15      }
16  
17      /**
18       * Metodi joka veloittaa pelaajalta konstruktorin määräämän summan jokaiselta hänen
19       * omistamaltaan mökiltä ja liikkeeltä
20       *
21       * @param pelaaja Pelaaja jolta veloitetaan rahaa
22       * 
23       * @return True jos rahojen veloitus onnistui, muuten false.
24       */
25      @Override
26      public boolean suoritaTapahtuma(Pelaaja pelaaja) {
27          int maksettavaSumma = 0
28                  + pelaaja.getOmistamatLiikkeet().size() * maksettavaHinta
29                  + pelaaja.getOmistamatMokit().size() * maksettavaHinta;
30  
31          Tapahtuma tapahtuma = new RahaanVaikuttavaTapahtuma(false, maksettavaSumma);
32          return tapahtuma.suoritaTapahtuma(pelaaja);
33      }
34  
35  }
