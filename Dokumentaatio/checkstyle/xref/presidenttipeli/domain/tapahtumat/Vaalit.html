1   package presidenttipeli.domain.tapahtumat;
2   
3   import java.util.ArrayList;
4   import java.util.Collections;
5   import presidenttipeli.domain.Pelaaja;
6   
7   /**
8    * Tapahtumaluokka joka suorittaa eduskunta- tai presidentinvaalit.
9    */
10  public class Vaalit implements Tapahtuma {
11  
12      private int tarvittavaAanimaara;
13      private ArrayList<Integer> aanet;
14      private ArrayList<Integer> saadutAanet;
15  
16      public Vaalit(int tarvittavaAanimaara) {
17          this.tarvittavaAanimaara = tarvittavaAanimaara;
18          aanet = new ArrayList();
19          saadutAanet = new ArrayList();
20          aanet.add(1);
21          aanet.add(3);
22          aanet.add(5);
23          aanet.add(7);
24          aanet.add(8);
25          aanet.add(9);
26          aanet.add(10);
27          aanet.add(12);
28          aanet.add(15);
29          aanet.add(20);
30          aanet.add(25);
31          aanet.add(30);
32          aanet.add(40);
33          aanet.add(50);
34      }
35  
36      /**
37       * Metodi joka palauttaa onnistuivatko vaalit. Vaalit tapahtuvat siten että
38       * otetaan 5 satunnaista "korttia" joissa lukee kunkin kortin äänimäärä,
39       * jotka lasketaan yhteen kokonaisäänimäärän saamiseksi.
40       *
41       *
42       * @param pelaaja Pelaaja jolle tapahtuma suoritetaan.
43       * 
44       * @return True jos vaalit onnistuivat, muuten false.
45       */
46      @Override
47      public boolean suoritaTapahtuma(Pelaaja pelaaja) {
48          return onnistuikoVaalit(laskeAanet());
49      }
50  
51      private int laskeAanet() {
52          Collections.shuffle(aanet);
53          int summa = 0;
54          for (int i = 0; i < 5; i++) {
55              saadutAanet.add(aanet.get(i));
56              summa += aanet.get(i);
57          }
58          return summa;
59      }
60  
61      /**
62       * Metodi joka tarkistaa onko saatu äänimäärä vähintään yhtäsuurikuin
63       * konstruktorin asettama kokonaislukumuuttuja tarvittavaAanimaara
64       *
65       * @param aanimaara saatu äänimäärä.
66       *
67       * @return True jos aanimaara onsuurempi tai yhtäsuurikuin
68       * tarvittavaAanimaara, muuten false.
69       */
70      public boolean onnistuikoVaalit(int aanimaara) {
71          return aanimaara >= tarvittavaAanimaara;
72      }
73  
74      /**
75       * Metodi joka palauttaa pelaajan saamat yksittäiset ääne.
76       *
77       * @return Pelaajan saamat äänet listana.
78       */
79      public ArrayList<Integer> getSaadutAanet() {
80          return saadutAanet;
81      }
82  
83  }
