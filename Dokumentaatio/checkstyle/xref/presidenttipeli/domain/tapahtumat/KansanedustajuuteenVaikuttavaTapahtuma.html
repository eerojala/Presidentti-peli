1   package presidenttipeli.domain.tapahtumat;
2   
3   import presidenttipeli.domain.Ammatti;
4   import presidenttipeli.domain.Pelaaja;
5   import presidenttipeli.domain.Pelilauta;
6   
7   /**
8    * Tapahtumaluokka joka vaikuttaa pelaajan kansanedustajuuteen.
9    */
10  public class KansanedustajuuteenVaikuttavaTapahtuma implements Tapahtuma {
11  
12      private boolean pelaajalleKansanedustajuus;
13      private Pelilauta lauta;
14  
15      public KansanedustajuuteenVaikuttavaTapahtuma(boolean pelaajalleKansanedustajuus, Pelilauta lauta) {
16          this.pelaajalleKansanedustajuus = pelaajalleKansanedustajuus;
17          this.lauta = lauta;
18      }
19  
20      /**
21       * Metodi antaa pelaajalle kansanedustajuuden tai ottaa sen pois riippuen
22       * konstrukorille annetusta totuusarvosta.
23       * 
24       * Jos Pelaaja saa kansanedustajuuden kun on ammatissa joka ei tätä salli,
25       * otetaan pelaajalta nykyinen ammatti pois ja asetetaan hänet työttömäksi.
26       *
27       * @param pelaaja Pelaaja jonka kansanedustajuuteen vaikutetaan.
28       * 
29       * @return Aina true.
30       */
31      @Override
32      public boolean suoritaTapahtuma(Pelaaja pelaaja) {
33          pelaaja.setKansanedustaja(pelaajalleKansanedustajuus);
34          if (pelaajalleKansanedustajuus == true) {
35              if (!pelaaja.getAmmatti().salliiKansanedustajuuden()) {
36                  Tapahtuma tapahtuma = new AsetaAmmatti(lauta, new Ammatti("Työtön", 1000, false, false, true));
37                  tapahtuma.suoritaTapahtuma(pelaaja);
38              }
39          }
40          return true;
41      }
42  
43  }
