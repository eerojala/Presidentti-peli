1   package presidenttipeli.domain.tapahtumat;
2   
3   import java.util.Collections;
4   import presidenttipeli.domain.Mokki;
5   import presidenttipeli.domain.Pelaaja;
6   import presidenttipeli.domain.Pelilauta;
7   
8   /**
9    * Tapahtumaluokka joka ottaa pelaajalta pois hänen arvokkaimman mökkinsä.
10   */
11  public class MenetaMokki implements Tapahtuma {
12  
13      private Pelilauta lauta;
14      private boolean vakuutusKorvaa;
15  
16      public MenetaMokki(Pelilauta lauta, boolean vakuutusKorvaa) {
17          this.lauta = lauta;
18          this.vakuutusKorvaa = vakuutusKorvaa;
19      }
20  
21      /**
22       * Metodi ottaa pelaajalta pois hänen arvokkaimman mökkinsä ja antaa
23       * korvauksena puolet sen arvosta mikäli konstruktorissa on asetettu muuttuja
24       * vakuutusKorvaa trueksi.
25       *
26       * @param pelaaja Pelaaja jolta otetaan hänen arvokkain mökkinsä pois
27       * 
28       * @return Aina true
29       */
30      @Override
31      public boolean suoritaTapahtuma(Pelaaja pelaaja) {
32          if (!pelaaja.getOmistamatMokit().isEmpty()) {
33              Collections.sort(pelaaja.getOmistamatMokit());
34              Mokki mokki = pelaaja.getOmistamatMokit().get(0);
35              Tapahtuma tapahtuma = new MokkeihinVaikuttavaTapahtuma(mokki, false, lauta);
36              tapahtuma.suoritaTapahtuma(pelaaja);
37  
38              if (vakuutusKorvaa) {
39                  tapahtuma = new RahaanVaikuttavaTapahtuma(true, mokki.getArvo() / 2);
40                  tapahtuma.suoritaTapahtuma(pelaaja);
41              }
42          }
43          return true;
44      }
45  
46  }
