1   package presidenttipeli.domain.tapahtumat;
2   
3   import java.util.Random;
4   import presidenttipeli.domain.Ammatti;
5   import presidenttipeli.domain.Pelaaja;
6   import presidenttipeli.domain.Pelilauta;
7   
8   /**
9    * Tapahtumaluokka joka ottaa laudalta johtaja- tai sattuma-ammatin.
10   */
11  public class OtaAmmattikortti implements Tapahtuma {
12  
13      private Pelilauta lauta;
14      private boolean johtaja; // jos true niin otetaan johtaja-ammatti, jos false niin sattuma-ammatti
15  
16      public OtaAmmattikortti(Pelilauta lauta, boolean johtaja) {
17          this.lauta = lauta;
18          this.johtaja = johtaja;
19      }
20  
21      /**
22       * Metodi ottaa joko johtaja- tai sattuma-ammatin pinosta ja asettaa sen
23       * pelaajan ammatiksi
24       *
25       * @param pelaaja Pelaaja jolle annetaan ammatti.
26       * 
27       * @return True jos AsetaAmmatti-tapahtuma onnistui, muuten false.
28       */
29      @Override
30      public boolean suoritaTapahtuma(Pelaaja pelaaja) {
31          Ammatti ammatti;
32          if (johtaja) {
33              ammatti = otaJohtaja();
34          } else {
35              ammatti = lauta.getSattumaAmmatit().pollFirst();
36          }
37          Tapahtuma tapahtuma = new AsetaAmmatti(lauta, ammatti);
38          return tapahtuma.suoritaTapahtuma(pelaaja);
39      }
40  
41      private Ammatti otaJohtaja() { // Ottaa satunnaisen johtaja-ammatin
42          Random random = new Random();
43          int indeksi = random.nextInt(lauta.getJohtajaAmmatit().size());
44          Ammatti ammatti = lauta.getJohtajaAmmatit().get(indeksi);
45          lauta.getJohtajaAmmatit().remove(indeksi); 
46          return ammatti;
47      }
48  
49  }
