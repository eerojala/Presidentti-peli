1   package presidenttipeli.domain.tapahtumat;
2   
3   import presidenttipeli.domain.Ammatti;
4   import presidenttipeli.domain.Pelaaja;
5   import presidenttipeli.domain.Pelilauta;
6   
7   /**
8    * Tapahtumaluokka joka asettaa pelaajalle uuden ammattin
9    */
10  public class AsetaAmmatti implements Tapahtuma {
11  
12      private Pelilauta lauta;
13      private Ammatti ammatti;
14  
15      public AsetaAmmatti(Pelilauta lauta, Ammatti ammatti) {
16          this.lauta = lauta;
17          this.ammatti = ammatti;
18      }
19  
20      /**
21       * Metodi asettaa pelaajalle konstruktorin määrittämän ammatin ja vie hänen 
22       * edellisen ammattinsavtakaisin pelilaudalle mikäli se oli johtaja- tai 
23       * sattuma-ammatti.
24       * Pelaajalta otetaan pois kansanedustajuus jos uusi ammatti ei salli sitä
25       * ja tutkintokin lähtee pois jos kyseessä ei ole yleissivistävä tutkinto.
26       *
27       * @param pelaaja Pelaaja jolle uusi ammatti asetetaan
28       * 
29       * @return Aina true
30       */
31      @Override
32      public boolean suoritaTapahtuma(Pelaaja pelaaja) {
33          Tapahtuma tapahtuma1 = new VieAmmattikorttiTakaisinPinoon(lauta, pelaaja.getAmmatti());
34          tapahtuma1.suoritaTapahtuma(pelaaja);
35          pelaaja.setAmmatti(ammatti);
36          otaPoisKansanedustajuus(pelaaja);
37          otaPoisTutkinto(pelaaja);
38          return true;
39      }
40  
41      private void otaPoisKansanedustajuus(Pelaaja pelaaja) {
42          if (pelaaja.isKansanedustaja() && ammatti.salliiKansanedustajuuden() == false) {
43              Tapahtuma tapahtuma = new KansanedustajuuteenVaikuttavaTapahtuma(false, lauta);
44              tapahtuma.suoritaTapahtuma(pelaaja);
45          }
46      }
47  
48      private void otaPoisTutkinto(Pelaaja pelaaja) {
49          if (pelaaja.getTutkinto() == null) {
50              return;
51          }
52  
53          if (!pelaaja.getTutkinto().isYleissivistava() && !ammatti.getNimi().equals("Työtön")) {
54              Tapahtuma tapahtuma = new TutkintoonVaikuttavaTapahtuma(false, false);
55              tapahtuma.suoritaTapahtuma(pelaaja);
56          }
57      }
58  
59  }
