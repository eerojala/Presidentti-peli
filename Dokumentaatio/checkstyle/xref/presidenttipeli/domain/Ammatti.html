3   /**
4    * Luokka pitää muistissa ammatin nimen ja palkan sekä tiedon onko ammatti
5    * johtaja-ammatti tai sattuma-ammatti ja salliiko se kansanedustajuuden.
6    */
7   public class Ammatti implements Comparable<Ammatti> {
8   
9       private String nimi;
10      private int palkka;
11      private boolean johtaja;
12      private boolean sattumaAmmmatti;
13      private boolean salliiKansanedustajuuden;
14  
15      public Ammatti(String nimi, int palkka, boolean johtaja, boolean sattumaAmmmatti, boolean salliiKansanedustajuuden) {
16          this.nimi = nimi;
17          this.palkka = palkka;
18          this.johtaja = johtaja;
19          this.sattumaAmmmatti = sattumaAmmmatti;
20          this.salliiKansanedustajuuden = salliiKansanedustajuuden;
21      }
22  
23      public String getNimi() {
24          return nimi;
25      }
26  
27      public int getPalkka() {
28          return palkka;
29      }
30  
31      public boolean isJohtaja() {
32          return johtaja;
33      }
34  
35      public boolean isSattumaAmmmatti() {
36          return sattumaAmmmatti;
37      }
38  
39      public boolean salliiKansanedustajuuden() {
40          return salliiKansanedustajuuden;
41      }
42  
43      public void setNimi(String nimi) {
44          this.nimi = nimi;
45      }
46  
47      public void setPalkka(int palkka) {
48          this.palkka = palkka;
49      }
50  
51      /**
52       * Metodi joka tarkastaa ovatko kaksi oliota samoja.
53       *
54       * @param obj Toinen olio
55       *
56       * @return Ovatko verrattavat oliot samoja (Molemmat ammatteja, samat nimet)
57       */
58      @Override
59      public boolean equals(Object obj) {
60          if (obj == null) {
61              return false;
62          }
63  
64          if (this.getClass() != obj.getClass()) {
65              return false;
66          }
67  
68          Ammatti verrattava = (Ammatti) obj;
69  
70          if (!this.nimi.equals(verrattava.getNimi())) {
71              return false;
72          }
73  
74          // Samannimisiä ammatteja ei voi olla
75          return true;
76      }
77  
78      /**
79       * Metodi joka vertaa kahta Ammatti-oliota toisiinsa
80       *
81       * @param o toinen ammatti
82       *
83       * @return Numero joka määrää ammattien keskenäisen järjestyksen
84       */
85      @Override
86      public int compareTo(Ammatti o) {
87          return o.getPalkka() - this.palkka;
88      }
89  
90      @Override
91      public String toString() {
92          return nimi + " , palkka: " + palkka + " mk/kk";
93      }
94      
95      
96  }
