309     private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
310         nimet.clear();
311         lisaaNimetListaan();
312         boolean samuustarkastus = aloitus.samuustarkastus(nimet);
313         boolean pituustarkastus = aloitus.pituustarkastus(nimet);
314         if (samuustarkastus == false || pituustarkastus == false) {
315             JOptionPane.showMessageDialog(rootPane, "Pelaajilla ei saa olla samoja nimiä ja niiden"
316                     + " pituuden tulisi olla 1-24 merkkiä", "Väärä syöte",
317                     JOptionPane.WARNING_MESSAGE);
318         } else {
319            KaikenLuoja luoja = new KaikenLuoja(nimet);
320            PeliGUI peligui = new PeliGUI(luoja.getPeli());
321            this.dispose();
322         }
323     }//GEN-LAST:event_jButton1ActionPerformed
