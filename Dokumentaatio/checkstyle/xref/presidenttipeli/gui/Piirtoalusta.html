1   package presidenttipeli.gui;
2   
3   import java.awt.Color;
4   import java.awt.Graphics;
5   import java.util.ArrayList;
6   import javax.swing.JPanel;
7   import presidenttipeli.domain.Nappula;
8   import presidenttipeli.domain.Pelilauta;
9   
10  /**
11   *  JPanel luokka joka piirtää nykyisen tilanteen pelilaudalla.
12   */
13  public class Piirtoalusta extends JPanel {
14  
15      private Pelilauta lauta;
16      private ArrayList<Nappula> nappulat;
17  
18      public Piirtoalusta(Pelilauta lauta) {
19          super.setBackground(Color.WHITE);
20          this.lauta = lauta;
21          nappulat = lauta.getNappulat();
22      }
23  
24      @Override
25      protected void paintComponent(Graphics graphics) {
26          super.paintComponent(graphics);
27          piirraLauta(graphics);
28          piirraNappulat(graphics);
29      }
30  
31      private void piirraLauta(Graphics graphics) {
32          graphics.drawImage(lauta.getKuva(), 0, 0, this);
33      }
34  
35      private void piirraNappulat(Graphics graphics) {
36          for (Nappula nappula : lauta.getNappulat()) {
37              graphics.drawImage(nappula.getKuva(), nappula.getSijainti().getX(),
38                      nappula.getSijainti().getY(), this);
39          }
40  
41      }
42  }
