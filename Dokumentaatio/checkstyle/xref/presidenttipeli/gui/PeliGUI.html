208     /**
209      * Ilmoittaa vaalien tuloksen dialogissa.
210      *
211      * @param onnistui Totuusarvo joka kertoo onnistuiko vaalit
212      * @param saadutAanet Saadut äänet yksittäisinä listattuna
213      * @param saadutAanetSummattuna Saadut äänet yhteensä
214      * @param tarvittavaAanimaara Vaalien voittoon tarvittu äänimäärä.
215      */
216     public void ilmoitaTulos(boolean onnistui, ArrayList<Integer> saadutAanet,
217             int saadutAanetSummattuna, int tarvittavaAanimaara) {
218         String title;
219 
220         if (onnistui) {
221             title = "Voitit vaaleissa!";
222         } else {
223             title = "Et voittanut vaaleissa";
224         }
225 
226         JOptionPane.showMessageDialog(rootPane,
227                 "Saamasi äänet: " + saadutAanet + "\n Yhteensä: " + saadutAanetSummattuna
228                 + "\n Tarvittava äänimäärä: " + tarvittavaAanimaara,
229                 title, JOptionPane.PLAIN_MESSAGE);
230     }
